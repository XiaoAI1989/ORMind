```python
from pulp import LpProblem, LpMaximize, LpVariable, value, PULP_CBC_CMD

def prob_0(DogCapability, TruckCapability, DogCost, TruckCost, MaxBudget):
    """
    Args:
        DogCapability: an integer, indicates the amount of fish which sled dogs can take per trip
        TruckCapability: an integer, indicates the amount of fish which trucks can take per trip
        DogCost: an integer, indicates the cost per trip for a sled dog
        TruckCost: an integer, indicates the cost per trip for a truck
        MaxBudget: an integer, denotes the upper limit of the budget

    Returns:
        FishTransported: an integer, denotes the amount of fish transported after calculation
    """
    prob = LpProblem("Fishery_Transport", LpMaximize)
    num_sled_dog_trips = LpVariable("sled_dog_trips", lowBound=0, cat='Integer')
    num_truck_trips = LpVariable("truck_trips", lowBound=0, cat='Integer')

    prob += (DogCapability * num_sled_dog_trips + TruckCapability * num_truck_trips)
    prob += (DogCost * num_sled_dog_trips + TruckCost * num_truck_trips <= MaxBudget)
    prob += (num_sled_dog_trips <= num_truck_trips - 1)

    prob.solve(PULP_CBC_CMD(msg=0))

    obj = value(prob.objective)
    var1 = int(value(num_sled_dog_trips))
    var2 = int(value(num_truck_trips))

    return obj, var1, var2
```
